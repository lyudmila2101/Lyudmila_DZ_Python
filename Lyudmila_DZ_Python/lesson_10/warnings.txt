# Установка зависимостей
pip install allure-pytest selenium webdriver-manager

# Запуск тестов
pytest --alluredir=allure-results

# Просмотр отчета
allure serve allure-results

# Функции можно документировать — описывать их логику. Используйте для описания три пары двойных кавычек:""" """

# allure generate название папки (allure-result)
# Результат выгрузится в папку allure-report.
